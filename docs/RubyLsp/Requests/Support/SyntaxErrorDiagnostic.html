def to_lsp_diagnostic
  LanguageServer::Protocol::Interface::Diagnostic.new(
    message: "Syntax error",
    source: "SyntaxTree",
    severity: LanguageServer::Protocol::Constant::DiagnosticSeverity::ERROR,
    range: @edit[:range]
  )
end
